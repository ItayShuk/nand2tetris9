Class Food{

    field int x;
    field int y;
    field int cols;
    field int rows;

    /** constructor of the class, gets the coordinates of where to draw the foo */
    Constructor Food new(Snake snake){
        let cols = 512;
        let rows = 256;
        let valid = 0;
        let i = 0;
        while (valid = 0){
            let x = random(4,cols-4);
            let y = random(4,rows-4);
            if (body_check(x,y,snake)){
                    let valid = 1;
            }
        }
        do draw();
        return this;
    }

    method int getX(){
        return x;
    }

    method int getY(){
        return y;
    }

    method int random(int a, int b) {
        var int d;
        var int r;
        let d = Math.divide(a,b);
        let r = a - (b * d);
        return Math.abs(r);
    }

    /** draws the food to the screen*/
    method void draw(){
        var int size;
        let size = 8;
        do Screen.setColor(true);
        do Screen.drawRectangle(x, y, x + size, y + size);
        return;
    }

    /** erases the food from the screen */
    method void erase(){
        var int size;
        let size = 8;
        do Screen.setColor(false);
        do Screen.drawRectangle(x, y, x + size, y + size);
        return;
    }

    /** disposes of the Food object */
    method void dispose(){
        do Memory.deAlloc(this);
        return;
    }

    method boolean body_check(int x,int y,Snake snake){
        let i = 0;
        while(i<snake.get_size()){
            if (x+4>snake.get_body()[i].getX() & x-4<snake.get_body()[i].getX()){
                if (y+4>snake.get_body()[i].getY() & y-4<snake.get_body()[i].getY()){
                    return false;
                }
            }
        }
        return true;
    }
}
