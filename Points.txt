Class Points{

    field int x;
    field int y;

    Constructor Points(int p_x, int p_y){
        let x = x;
        let y = y;
    }

    method int getX(){
        return x;
    }

    method int getY(){
        return y;
    }

    method void setX(int num){
        let x = num;
        return;
    }

    method void setY(int num){
        let y = num;
        return;
    }
}
